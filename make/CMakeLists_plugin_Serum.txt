# plugin CMake file to be included in main project CMakelists

# --- Serum colorization plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(BUILD_LIBS_DIR "${CMAKE_SOURCE_DIR}/third-party/build-libs/${PLATFORM}-${ARCH}")
set(RUNTIME_LIBS_DIR "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PLATFORM}-${ARCH}")

set(SERUM_PLUGIN_SOURCES
   plugins/serum/common.cpp
   plugins/serum/serum.cpp
)

if(BUILD_SHARED)
   add_library(SerumPlugin MODULE ${SERUM_PLUGIN_SOURCES})

   target_include_directories(SerumPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/serum
   )

   if(PLATFORM STREQUAL "windows" OR PLATFORM STREQUAL "windows-mingw")
      if(MSVC)
         set_target_properties(SerumPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>"
         )
         set(OPT_COMMON
            /Ob2
            /O2
            /Oi
            /fp:fast
            /fp:except-
            /Ot
            /GF
            /GS-
            /Gy
            /GR
            /Oy
            /GT
            /GL
            /w44005
            /Zc:__cplusplus
            /std:c++20
            $<$<NOT:$<STREQUAL:${ARCH},x64>>:/arch:SSE2>
         )
         target_compile_options(SerumPlugin PUBLIC
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(SerumPlugin PUBLIC
            $<$<CONFIG:Release>:/INCREMENTAL:NO>
            $<$<CONFIG:Release>:/OPT:REF>
            $<$<CONFIG:Release>:/OPT:ICF>
            $<$<CONFIG:Release>:/LTCG>
         )
      endif()
      target_link_directories(SerumPlugin PUBLIC ${BUILD_LIBS_DIR} ${RUNTIME_LIBS_DIR})
      set_target_properties(SerumPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "$<IF:$<STREQUAL:${ARCH},x64>,plugin-serum64,plugin-serum>"
      )
      target_link_libraries(SerumPlugin $<IF:$<STREQUAL:${ARCH},x64>,serum64,serum>)
   else()
      set_target_properties(SerumPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-serum"
      )
      if(PLATFORM STREQUAL "macos")
         set_target_properties(SerumPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
      target_link_directories(SerumPlugin PUBLIC
         ${RUNTIME_LIBS_DIR}
      )
      target_link_libraries(SerumPlugin PUBLIC serum)
   endif()

   add_custom_command(TARGET SerumPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/serum/plugin.cfg" "${PLUGINS_DIR}/serum/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:SerumPlugin>" "${PLUGINS_DIR}/serum"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:SerumPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS)
      if(PLATFORM STREQUAL "windows")
         if (ARCH STREQUAL "x64")
            add_custom_command(TARGET SerumPlugin POST_BUILD
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/serum64.dll" "${PLUGINS_DIR}/serum/serum64.dll"
            )
         elseif (ARCH STREQUAL "x86")
            add_custom_command(TARGET SerumPlugin POST_BUILD
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/serum.dll" "${PLUGINS_DIR}/serum/serum.dll"
            )
         endif()
      elseif(PLATFORM STREQUAL "windows-mingw")
         add_custom_command(TARGET SerumPlugin POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/serum64.dll" "${PLUGINS_DIR}/serum/serum64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/serum/libgcc_s_seh-1.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libstdc++-6.dll" "${PLUGINS_DIR}/serum/libstdc++-6.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libwinpthread-1.dll" "${PLUGINS_DIR}/serum/libwinpthread-1.dll"
         )
      elseif(PLATFORM STREQUAL "linux")
         add_custom_command(TARGET SerumPlugin POST_BUILD
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libserum.so" "${PLUGINS_DIR}/serum/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libserum.so.*" "${PLUGINS_DIR}/serum/"
         )
      endif()
   endif()
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(SerumPlugin STATIC ${SERUM_PLUGIN_SOURCES})

   target_include_directories(SerumPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/serum
   )

   set_target_properties(SerumPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-serum"
   )
endif()
