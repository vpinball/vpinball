# plugin CMake file to be included in main project CMakelists

# --- Serum colorization plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(SERUM_PLUGIN_SOURCES
   plugins/serum/common.cpp
   plugins/serum/serum.cpp
)

if(BUILD_SHARED)
   add_library(SerumPlugin MODULE ${SERUM_PLUGIN_SOURCES})

   target_include_directories(SerumPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/serum
   )

   if(PluginPlatform STREQUAL "windows" OR PluginPlatform STREQUAL "windows-mingw")
      if(MSVC)
         add_compile_options(
            $<$<CONFIG:RELEASE>:/Ob2>
            $<$<CONFIG:RELEASE>:/O2>
            $<$<CONFIG:RELEASE>:/Oi>
            $<$<CONFIG:RELEASE>:/arch:SSE2>
            $<$<CONFIG:RELEASE>:/fp:fast>
            $<$<CONFIG:RELEASE>:/fp:except->
            $<$<CONFIG:RELEASE>:/Ot>
            $<$<CONFIG:RELEASE>:/GF>
            $<$<CONFIG:RELEASE>:/GS->
            $<$<CONFIG:RELEASE>:/Gy>
            $<$<CONFIG:RELEASE>:/GR>
            $<$<CONFIG:RELEASE>:/Oy>
            $<$<CONFIG:RELEASE>:/GT>
            $<$<CONFIG:RELEASE>:/GL>
         )
         set_target_properties(SerumPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:DEBUG>:Debug>"
         )
         target_compile_options(SerumPlugin PUBLIC
            $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(SerumPlugin PUBLIC
            $<$<CONFIG:RELEASE>:/INCREMENTAL:NO>
            $<$<CONFIG:RELEASE>:/OPT:REF>
            $<$<CONFIG:RELEASE>:/OPT:ICF>
            $<$<CONFIG:RELEASE>:/LTCG>
         )
      endif()
      target_link_directories(SerumPlugin PUBLIC ${CMAKE_SOURCE_DIR}/third-party/build-libs/${PluginPlatform}-${PluginArch} ${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch})
      if (PluginArch STREQUAL "x64")
         set_target_properties(SerumPlugin PROPERTIES
            PREFIX ""
            OUTPUT_NAME "plugin-serum64"
         )
         target_link_libraries(SerumPlugin serum64)
      else()
         set_target_properties(SerumPlugin PROPERTIES
            PREFIX ""
            OUTPUT_NAME "plugin-serum"
         )
         target_link_libraries(SerumPlugin serum)
      endif()
   else()
      set_target_properties(SerumPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-serum"
      )
      if(PluginPlatform STREQUAL "macos")
         set_target_properties(SerumPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
      target_link_directories(SerumPlugin PUBLIC
         ${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}
      )
      target_link_libraries(SerumPlugin PUBLIC serum)
   endif()

   add_custom_command(TARGET SerumPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/serum/plugin.cfg" "${PLUGINS_DIR}/serum/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:SerumPlugin>" "${PLUGINS_DIR}/serum"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:SerumPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS)
      if(PluginPlatform STREQUAL "windows")
         if (PluginArch STREQUAL "x64")
            add_custom_command(TARGET SerumPlugin POST_BUILD
               COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/serum64.dll" "${PLUGINS_DIR}/serum/serum64.dll"
            )
         elseif (PluginArch STREQUAL "x86")
            add_custom_command(TARGET SerumPlugin POST_BUILD
               COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/serum.dll" "${PLUGINS_DIR}/serum/serum.dll"
            )
         endif()
      elseif(PluginPlatform STREQUAL "windows-mingw")
         add_custom_command(TARGET SerumPlugin POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/serum64.dll" "${PLUGINS_DIR}/serum/serum64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/serum/libgcc_s_seh-1.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/libstdc++-6.dll" "${PLUGINS_DIR}/serum/libstdc++-6.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}/libwinpthread-1.dll" "${PLUGINS_DIR}/serum/libwinpthread-1.dll"
         )
      elseif(PluginPlatform STREQUAL "linux")
         add_custom_command(TARGET SerumPlugin POST_BUILD
            COMMAND cp -a "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/linux-${PluginArch}/libserum.so" "${PLUGINS_DIR}/serum/"
            COMMAND cp -a "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/linux-${PluginArch}/libserum.so.*" "${PLUGINS_DIR}/serum/"
         )
      endif()
   endif()
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(SerumPlugin STATIC ${SERUM_PLUGIN_SOURCES})

   target_include_directories(SerumPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/serum
   )

   set_target_properties(SerumPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-serum"
   )
endif()
