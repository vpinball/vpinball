if(BUILD_SHARED)
   add_library(B2SLegacyPlugin MODULE ${B2SLEGACY_PLUGIN_SOURCES})

   target_include_directories(B2SLegacyPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/b2slegacy
      src/core
      src
   )

   if(PLATFORM STREQUAL "windows" OR PLATFORM STREQUAL "windows-mingw")
      if(MSVC)
         set_target_properties(B2SLegacyPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>"
         )
         set(OPT_COMMON
            /Ob2
            /O2
            /Oi
            /fp:fast
            /fp:except-
            /Ot
            /GF
            /GS-
            /Gy
            /GR
            /Oy
            /GT
            /GL
            /w44005
            /Zc:__cplusplus
            /std:c++20
            $<$<NOT:$<STREQUAL:${ARCH},x64>>:/arch:SSE2>
         )
         target_compile_options(B2SLegacyPlugin PUBLIC
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(B2SLegacyPlugin PUBLIC
            $<$<CONFIG:Release>:/INCREMENTAL:NO>
            $<$<CONFIG:Release>:/OPT:REF>
            $<$<CONFIG:Release>:/OPT:ICF>
            $<$<CONFIG:Release>:/LTCG>
         )
      endif()
      target_link_directories(B2SLegacyPlugin PUBLIC ${BUILD_LIBS_DIR} ${RUNTIME_LIBS_DIR})
      set_target_properties(B2SLegacyPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "$<IF:$<STREQUAL:${ARCH},x64>,plugin-b2slegacy64,plugin-b2slegacy>"
      )
      target_link_libraries(B2SLegacyPlugin $<IF:$<STREQUAL:${ARCH},x64>,SDL364,SDL3>)
   else()
      set_target_properties(B2SLegacyPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-b2slegacy"
      )
      if(PLATFORM STREQUAL "macos")
         set_target_properties(B2SLegacyPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
      target_link_directories(B2SLegacyPlugin PUBLIC
         ${RUNTIME_LIBS_DIR}
      )
      target_link_libraries(B2SLegacyPlugin PUBLIC SDL3)
   endif()

   add_custom_command(TARGET B2SLegacyPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/b2slegacy/plugin.cfg" "${PLUGINS_DIR}/b2slegacy/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:B2SLegacyPlugin>" "${PLUGINS_DIR}/b2slegacy"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:B2SLegacyPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS)
      if(PLATFORM STREQUAL "windows")
         if (ARCH STREQUAL "x64")
            add_custom_command(TARGET B2SLegacyPlugin POST_BUILD
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/SDL364.dll" "${PLUGINS_DIR}/b2slegacy/SDL364.dll"
            )
         elseif (ARCH STREQUAL "x86")
            add_custom_command(TARGET B2SLegacyPlugin POST_BUILD
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/SDL3.dll" "${PLUGINS_DIR}/b2slegacy/SDL3.dll"
            )
         endif()
      elseif(PLATFORM STREQUAL "windows-mingw")
         add_custom_command(TARGET B2SLegacyPlugin POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/SDL364.dll" "${PLUGINS_DIR}/b2slegacy/SDL364.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/b2slegacy/libgcc_s_seh-1.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libstdc++-6.dll" "${PLUGINS_DIR}/b2slegacy/libstdc++-6.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libwinpthread-1.dll" "${PLUGINS_DIR}/b2slegacy/libwinpthread-1.dll"
         )
      elseif(PLATFORM STREQUAL "linux")
         add_custom_command(TARGET B2SLegacyPlugin POST_BUILD
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libSDL3.so" "${PLUGINS_DIR}/b2slegacy/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libSDL3.so.*" "${PLUGINS_DIR}/b2slegacy/"
         )
      endif()
   endif()
endif()

   add_compile_options(-fPIC)

   target_include_directories(B2SLegacyPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/b2slegacy
      src/core
      src
   )

   set_target_properties(B2SLegacyPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-b2slegacy"
   )
endif()
