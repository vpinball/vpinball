# --- PUP plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(BUILD_LIBS_DIR "${CMAKE_SOURCE_DIR}/third-party/build-libs/${PLATFORM}-${ARCH}")
set(RUNTIME_LIBS_DIR "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PLATFORM}-${ARCH}")

set(PUP_PLUGIN_SOURCES

if(BUILD_SHARED AND NOT (PLATFORM STREQUAL "windows" AND NOT MSVC))
   add_library(PUPPlugin MODULE ${PUP_PLUGIN_SOURCES})

   target_include_directories(PUPPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/pup
   )

   if(PLATFORM STREQUAL "windows" OR PLATFORM STREQUAL "windows-mingw")
      if(MSVC)
         set_target_properties(PUPPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>"
         )
         set(OPT_COMMON
            /Ob2
            /O2
            /Oi
            /fp:fast
            /fp:except-
            /Ot
            /GF
            /GS-
            /Gy
            /GR
            /Oy
            /GT
            /GL
            /w44005
            /Zc:__cplusplus
            /std:c++20
            $<$<NOT:$<STREQUAL:${ARCH},x64>>:/arch:SSE2>
         )
         target_compile_options(PUPPlugin PUBLIC
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(PUPPlugin PUBLIC
            $<$<CONFIG:Release>:/INCREMENTAL:NO>
            $<$<CONFIG:Release>:/OPT:REF>
            $<$<CONFIG:Release>:/OPT:ICF>
            $<$<CONFIG:Release>:/LTCG>
            /SAFESEH:NO
         )
      endif()
      target_link_directories(PUPPlugin PUBLIC ${BUILD_LIBS_DIR} ${RUNTIME_LIBS_DIR})
      set_target_properties(PUPPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "$<IF:$<STREQUAL:${ARCH},x64>,plugin-pup64,plugin-pup>"
      )
      target_link_libraries(PUPPlugin
         $<IF:$<STREQUAL:${ARCH},x64>,SDL364,SDL3>
         $<IF:$<STREQUAL:${ARCH},x64>,SDL3_image64,SDL3_image>
         $<IF:$<STREQUAL:${ARCH},x64>,SDL3_ttf64,SDL3_ttf>
         $<IF:$<STREQUAL:${ARCH},x64>,pupdmd64,pupdmd>
      )
   else()
      set_target_properties(PUPPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-pup"
      )
      if(PLATFORM STREQUAL "macos")
         set_target_properties(PUPPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
      target_link_directories(PUPPlugin PUBLIC
         ${RUNTIME_LIBS_DIR}
      )
      target_link_libraries(PUPPlugin PUBLIC
         SDL3
         SDL3_image
         SDL3_ttf
         avcodec
         avformat
         avutil
         swresample
         swscale
         pupdmd
      )
   endif()

   add_custom_command(TARGET PUPPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/pup/plugin.cfg" "${PLUGINS_DIR}/pup/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:PUPPlugin>" "${PLUGINS_DIR}/pup"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:PUPPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS)

               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/SDL3_ttf.dll" "${PLUGINS_DIR}/pup/SDL3_ttf.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libiconv-2.dll" "${PLUGINS_DIR}/pup/libiconv-2.dll"

      elseif(PLATFORM STREQUAL "windows-mingw")

      elseif(PLATFORM STREQUAL "linux")
         add_custom_command(TARGET PUPPlugin POST_BUILD
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libSDL3.so" "${PLUGINS_DIR}/pup/"

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(PUPPlugin STATIC ${PUP_PLUGIN_SOURCES})

   target_include_directories(PUPPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/pup
   )

   set_target_properties(PUPPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-pup"
   )
   target_link_directories(PUPPlugin PUBLIC
      ${RUNTIME_LIBS_DIR}
