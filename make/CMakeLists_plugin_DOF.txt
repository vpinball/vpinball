# plugin CMake file to be included in main project CMakelists

# --- DOF plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(DOF_PLUGIN_SOURCES
   plugins/dof/DOFPlugin.cpp
)

if(BUILD_SHARED)
   add_library(DOFPlugin MODULE ${DOF_PLUGIN_SOURCES})

   target_include_directories(DOFPlugin PUBLIC
      plugins/dof
      ${CMAKE_SOURCE_DIR}/third-party/include
      src/plugins
      src
   )

   if(PluginPlatform STREQUAL "windows")
      add_compile_options(
         $<$<CONFIG:RELEASE>:/Ob2>
         $<$<CONFIG:RELEASE>:/O2>
         $<$<CONFIG:RELEASE>:/Oi>
         $<$<CONFIG:RELEASE>:/arch:SSE2>
         $<$<CONFIG:RELEASE>:/fp:fast>
         $<$<CONFIG:RELEASE>:/fp:except->
         $<$<CONFIG:RELEASE>:/Ot>
         $<$<CONFIG:RELEASE>:/GF>
         $<$<CONFIG:RELEASE>:/GS->
         $<$<CONFIG:RELEASE>:/Gy>
         $<$<CONFIG:RELEASE>:/GR>
         $<$<CONFIG:RELEASE>:/Oy>
         $<$<CONFIG:RELEASE>:/GT>
         $<$<CONFIG:RELEASE>:/GL>
      )
      set_target_properties(DOFPlugin PROPERTIES
         MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:DEBUG>:Debug>"
      )
      target_compile_options(DOFPlugin PUBLIC
         $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
         $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
      )
      target_link_options(DOFPlugin PUBLIC
         $<$<CONFIG:RELEASE>:/INCREMENTAL:NO>
         $<$<CONFIG:RELEASE>:/OPT:REF>
         $<$<CONFIG:RELEASE>:/OPT:ICF>
         $<$<CONFIG:RELEASE>:/LTCG>
      )
      if (PluginArch STREQUAL "x64")
         set_target_properties(DOFPlugin PROPERTIES
            OUTPUT_NAME "plugin-dof64"
         )
         target_link_directories(DOFPlugin PUBLIC ${CMAKE_SOURCE_DIR}/third-party/build-libs/windows-x64)
         target_link_libraries(DOFPlugin dof64.lib)
      else()
         set_target_properties(DOFPlugin PROPERTIES
            OUTPUT_NAME "plugin-dof"
         )
         target_link_directories(DOFPlugin PUBLIC ${CMAKE_SOURCE_DIR}/third-party/build-libs/windows-x86)
         target_link_libraries(DOFPlugin dof.lib)
      endif()
   else()
      set_target_properties(DOFPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-dof"
      )
      if(PluginPlatform STREQUAL "macos")
         set_target_properties(DOFPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
      target_link_directories(DOFPlugin PUBLIC
         ${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PluginPlatform}-${PluginArch}
      )
      target_link_libraries(DOFPlugin PUBLIC dof)
   endif()

   add_custom_command(TARGET DOFPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/dof/plugin.cfg" "${PLUGINS_DIR}/dof/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:DOFPlugin>" "${PLUGINS_DIR}/dof"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:DOFPlugin>"
   )

   if(PluginPlatform STREQUAL "windows")
      if (PluginArch STREQUAL "x64")
         add_custom_command(TARGET DOFPlugin POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x64/dof64.dll" "${PLUGINS_DIR}/dof/dof64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x64/libusb64-1.0.dll" "${PLUGINS_DIR}/dof/libusb64-1.0.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x64/libserialport64.dll" "${PLUGINS_DIR}/dof/libserialport64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x64/hidapi64.dll" "${PLUGINS_DIR}/dof/hidapi64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x64/libftdi164.dll" "${PLUGINS_DIR}/dof/libftdi164.dll"
         )
      elseif (PluginArch STREQUAL "x86")
         add_custom_command(TARGET DOFPlugin POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x86/dof.dll" "${PLUGINS_DIR}/dof/dof.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x86/libusb-1.0.dll" "${PLUGINS_DIR}/dof/libusb-1.0.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x86/libserialport.dll" "${PLUGINS_DIR}/dof/libserialport.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x86/hidapi.dll" "${PLUGINS_DIR}/dof/hidapi.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/windows-x86/libftdi1.dll" "${PLUGINS_DIR}/dof/libftdi1.dll"
         )
      endif()
   endif()
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(DOFPlugin STATIC ${DOF_PLUGIN_SOURCES})

   target_include_directories(DOFPlugin PUBLIC
      plugins/dof
      ${CMAKE_SOURCE_DIR}/third-party/include
      src/plugins
      src
   )

   set_target_properties(DOFPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-dof"
   )
endif()
