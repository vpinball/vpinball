# plugin CMake file to be included in main project CMakelists

# --- DMDUtil colorization plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(BUILD_LIBS_DIR "${CMAKE_SOURCE_DIR}/third-party/build-libs/${PLATFORM}-${ARCH}")
set(RUNTIME_LIBS_DIR "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PLATFORM}-${ARCH}")

set(DMDUTIL_PLUGIN_SOURCES
   plugins/dmdutil/DMDUtilPlugin.cpp
)

if(BUILD_SHARED AND NOT (PLATFORM STREQUAL "windows" AND NOT MSVC))
   add_library(DMDUtilPlugin MODULE ${DMDUTIL_PLUGIN_SOURCES})

   target_include_directories(DMDUtilPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/dmdutil
   )

   if(PLATFORM STREQUAL "windows" OR PLATFORM STREQUAL "windows-mingw")
      if(MSVC)
         set_target_properties(DMDUtilPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>"
         )
         set(OPT_COMMON
            /Ob2
            /O2
            /Oi
            /fp:fast
            /fp:except-
            /Ot
            /GF
            /GS-
            /Gy
            /GR
            /Oy
            /GT
            /GL
            /w44005
            /Zc:__cplusplus
            /std:c++20
            $<$<NOT:$<STREQUAL:${ARCH},x64>>:/arch:SSE2>
         )
         target_compile_options(DMDUtilPlugin PUBLIC
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(DMDUtilPlugin PUBLIC
            $<$<CONFIG:Release>:/INCREMENTAL:NO>
            $<$<CONFIG:Release>:/OPT:REF>
            $<$<CONFIG:Release>:/OPT:ICF>
            $<$<CONFIG:Release>:/LTCG>
         )
      endif()
      target_link_directories(DMDUtilPlugin PUBLIC
         ${BUILD_LIBS_DIR}
         ${RUNTIME_LIBS_DIR}
      )
      set_target_properties(DMDUtilPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "$<IF:$<STREQUAL:${ARCH},x64>,plugin-dmdutil64,plugin-dmdutil>"
      )
      target_link_libraries(DMDUtilPlugin $<IF:$<STREQUAL:${ARCH},x64>,dmdutil64,dmdutil>)
   else()
      set_target_properties(DMDUtilPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-dmdutil"
      )
      if(PLATFORM STREQUAL "macos")
         set_target_properties(DMDUtilPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
      target_link_directories(DMDUtilPlugin PUBLIC
         ${RUNTIME_LIBS_DIR}
      )
      target_link_libraries(DMDUtilPlugin PUBLIC dmdutil)
   endif()

   add_custom_command(TARGET DMDUtilPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/dmdutil/plugin.cfg" "${PLUGINS_DIR}/dmdutil/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:DMDUtilPlugin>" "${PLUGINS_DIR}/dmdutil"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:DMDUtilPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS)
      if(PLATFORM STREQUAL "windows")
         if (ARCH STREQUAL "x64")
            add_custom_command(TARGET DMDUtilPlugin POST_BUILD
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/dmdutil64.dll" "${PLUGINS_DIR}/dmdutil/dmdutil64.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libusb64-1.0.dll" "${PLUGINS_DIR}/dmdutil/libusb64-1.0.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libserialport64-0.dll" "${PLUGINS_DIR}/dmdutil/libserialport64-0.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/sockpp64.dll" "${PLUGINS_DIR}/dmdutil/sockpp64.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/cargs64.dll" "${PLUGINS_DIR}/dmdutil/cargs64.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/zedmd64.dll" "${PLUGINS_DIR}/dmdutil/zedmd64.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/serum64.dll" "${PLUGINS_DIR}/dmdutil/serum64.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/pupdmd64.dll" "${PLUGINS_DIR}/dmdutil/pupdmd64.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/vni64.dll" "${PLUGINS_DIR}/dmdutil/vni64.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/dmdutil/libgcc_s_seh-1.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libstdc++-6.dll" "${PLUGINS_DIR}/dmdutil/libstdc++-6.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libwinpthread-1.dll" "${PLUGINS_DIR}/dmdutil/libwinpthread-1.dll"
            )
         elseif (ARCH STREQUAL "x86")
            add_custom_command(TARGET DMDUtilPlugin POST_BUILD
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/dmdutil.dll" "${PLUGINS_DIR}/dmdutil/dmdutil.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libusb-1.0.dll" "${PLUGINS_DIR}/dmdutil/libusb-1.0.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libserialport-0.dll" "${PLUGINS_DIR}/dmdutil/libserialport-0.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/sockpp.dll" "${PLUGINS_DIR}/dmdutil/sockpp.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/cargs.dll" "${PLUGINS_DIR}/dmdutil/cargs.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/zedmd.dll" "${PLUGINS_DIR}/dmdutil/zedmd.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/serum.dll" "${PLUGINS_DIR}/dmdutil/serum.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/pupdmd.dll" "${PLUGINS_DIR}/dmdutil/pupdmd.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/vni.dll" "${PLUGINS_DIR}/dmdutil/vni.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libgcc_s_dw2-1.dll" "${PLUGINS_DIR}/dmdutil/libgcc_s_dw2-1.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libstdc++-6.dll" "${PLUGINS_DIR}/dmdutil/libstdc++-6.dll"
               COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libwinpthread-1.dll" "${PLUGINS_DIR}/dmdutil/libwinpthread-1.dll"
            )
         endif()
      elseif(PLATFORM STREQUAL "windows-mingw")
         add_custom_command(TARGET DMDUtilPlugin POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/dmdutil64.dll" "${PLUGINS_DIR}/dmdutil/dmdutil64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libusb64-1.0.dll" "${PLUGINS_DIR}/dmdutil/libusb64-1.0.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libserialport64-0.dll" "${PLUGINS_DIR}/dmdutil/libserialport64-0.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libsockpp64.dll" "${PLUGINS_DIR}/dmdutil/libsockpp64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libcargs64.dll" "${PLUGINS_DIR}/dmdutil/libcargs64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/zedmd64.dll" "${PLUGINS_DIR}/dmdutil/zedmd64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/serum64.dll" "${PLUGINS_DIR}/dmdutil/serum64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/pupdmd64.dll" "${PLUGINS_DIR}/dmdutil/pupdmd64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/vni64.dll" "${PLUGINS_DIR}/dmdutil/vni64.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/dmdutil/libgcc_s_seh-1.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libstdc++-6.dll" "${PLUGINS_DIR}/dmdutil/libstdc++-6.dll"
            COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libwinpthread-1.dll" "${PLUGINS_DIR}/dmdutil/libwinpthread-1.dll"
         )
      elseif(PLATFORM STREQUAL "linux")
         add_custom_command(TARGET DMDUtilPlugin POST_BUILD
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libdmdutil.so" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libdmdutil.so.*" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libcargs.so" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libusb-1.0.so" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libusb-1.0.so.*" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libserialport.so" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libserialport.so.*" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libsockpp.so" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libsockpp.so.*" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libzedmd.so" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libzedmd.so.*" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libpupdmd.so" "${PLUGINS_DIR}/dmdutil/"

            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libserum.so" "${PLUGINS_DIR}/dmdutil/"
            COMMAND cp -a "${RUNTIME_LIBS_DIR}/libserum.so.*" "${PLUGINS_DIR}/dmdutil/"
         )
      endif()
   endif()
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(DMDUtilPlugin STATIC ${DMDUTIL_PLUGIN_SOURCES})

   target_include_directories(DMDUtilPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/dmdutil
   )

   set_target_properties(DMDUtilPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-dmdutil"
   )
endif()
