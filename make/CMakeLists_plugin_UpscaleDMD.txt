# plugin CMake file to be included in main project CMakelists

# --- DisplayDMD plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(ALPHADMD_PLUGIN_SOURCES
   plugins/displaydmd/displaydmd.cpp
)

if(BUILD_SHARED)
   add_library(DisplayDMDPlugin MODULE ${ALPHADMD_PLUGIN_SOURCES})

   target_include_directories(DisplayDMDPlugin PUBLIC
      plugins/displaydmd
      ${CMAKE_SOURCE_DIR}/third-party/include
      src/plugins
      src
   )

   if(PluginPlatform STREQUAL "windows")
      add_compile_options(
         $<$<CONFIG:RELEASE>:/Ob2>
         $<$<CONFIG:RELEASE>:/O2>
         $<$<CONFIG:RELEASE>:/Oi>
         $<$<CONFIG:RELEASE>:/arch:SSE2>
         $<$<CONFIG:RELEASE>:/fp:fast>
         $<$<CONFIG:RELEASE>:/fp:except->
         $<$<CONFIG:RELEASE>:/Ot>
         $<$<CONFIG:RELEASE>:/GF>
         $<$<CONFIG:RELEASE>:/GS->
         $<$<CONFIG:RELEASE>:/Gy>
         $<$<CONFIG:RELEASE>:/GR>
         $<$<CONFIG:RELEASE>:/Oy>
         $<$<CONFIG:RELEASE>:/GT>
         $<$<CONFIG:RELEASE>:/GL>
      )
      set_target_properties(DisplayDMDPlugin PROPERTIES
         MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:DEBUG>:Debug>"
      )
      target_compile_options(DisplayDMDPlugin PUBLIC
         $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
         $<$<CONFIG:RELEASE>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
      )
      target_link_options(DisplayDMDPlugin PUBLIC
         $<$<CONFIG:RELEASE>:/INCREMENTAL:NO>
         $<$<CONFIG:RELEASE>:/OPT:REF>
         $<$<CONFIG:RELEASE>:/OPT:ICF>
         $<$<CONFIG:RELEASE>:/LTCG>
      )
      if(PluginArch STREQUAL "x64")
         set_target_properties(DisplayDMDPlugin PROPERTIES
            OUTPUT_NAME "plugin-displaydmd64"
         )
      else()
         set_target_properties(DisplayDMDPlugin PROPERTIES
            OUTPUT_NAME "plugin-displaydmd"
         )
      endif()
   else()
      set_target_properties(DisplayDMDPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-displaydmd"
      )
      if(PluginPlatform STREQUAL "macos")
         set_target_properties(DisplayDMDPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
   endif()

   add_custom_command(TARGET DisplayDMDPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/displaydmd/plugin.cfg" "${PLUGINS_DIR}/displaydmd/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:DisplayDMDPlugin>" "${PLUGINS_DIR}/displaydmd"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:DisplayDMDPlugin>"
   )
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(DisplayDMDPlugin STATIC ${ALPHADMD_PLUGIN_SOURCES})

   target_include_directories(DisplayDMDPlugin PUBLIC
      plugins/displaydmd
      ${CMAKE_SOURCE_DIR}/third-party/include
      src/plugins
      src
   )

   set_target_properties(DisplayDMDPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-displaydmd"
   )
endif()
