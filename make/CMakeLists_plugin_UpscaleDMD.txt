# plugin CMake file to be included in main project CMakelists

# --- UpscaleDMD plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(RUNTIME_LIBS_DIR "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PLATFORM}-${ARCH}")

set(UPSCALEDMD_PLUGIN_SOURCES
   plugins/upscaledmd/UpscaleDMD.cpp
   plugins/upscaledmd/xbrz/xbrz.cpp
   plugins/upscaledmd/xbrz/xbrz.h
   plugins/upscaledmd/super-xbr/super-xbr.h
   plugins/upscaledmd/scalefx/scalefx.h
   plugins/plugins/ResURIResolver.h
   plugins/plugins/ResURIResolver.cpp
)

if(BUILD_SHARED)
   add_library(UpscaleDMDPlugin MODULE ${UPSCALEDMD_PLUGIN_SOURCES})

   target_include_directories(UpscaleDMDPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/upscaledmd
   )

   if(PLATFORM STREQUAL "windows" OR PLATFORM STREQUAL "windows-mingw")
      if(MSVC)
         set_target_properties(UpscaleDMDPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>"
         )
         set(OPT_COMMON
            /Ob2
            /O2
            /Oi
            /fp:fast
            /fp:except-
            /Ot
            /GF
            /GS-
            /Gy
            /GR
            /Oy
            /GT
            /GL
            /w44005
            /Zc:__cplusplus
            /std:c++20
            $<$<NOT:$<STREQUAL:${ARCH},x64>>:/arch:SSE2>
         )
         target_compile_options(UpscaleDMDPlugin PUBLIC
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(UpscaleDMDPlugin PUBLIC
            $<$<CONFIG:Release>:/INCREMENTAL:NO>
            $<$<CONFIG:Release>:/OPT:REF>
            $<$<CONFIG:Release>:/OPT:ICF>
            $<$<CONFIG:Release>:/LTCG>
         )
      endif()
      if(NOT MSVC AND ARCH STREQUAL "x64")
         target_compile_options(UpscaleDMDPlugin PUBLIC
            -mssse3
         )
      endif()
      set_target_properties(UpscaleDMDPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "$<IF:$<STREQUAL:${ARCH},x64>,plugin-upscaledmd64,plugin-upscaledmd>"
      )
   else()
      set_target_properties(UpscaleDMDPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-upscaledmd"
      )
      if(PLATFORM STREQUAL "macos")
         set_target_properties(UpscaleDMDPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
      if(PLATFORM STREQUAL "linux")
         if(ARCH STREQUAL "x64")
            target_compile_options(UpscaleDMDPlugin PUBLIC
               -mssse3
            )
         endif()
      endif()
   endif()

   add_custom_command(TARGET UpscaleDMDPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/upscaledmd/plugin.cfg" "${PLUGINS_DIR}/upscaledmd/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:UpscaleDMDPlugin>" "${PLUGINS_DIR}/upscaledmd"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:UpscaleDMDPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS AND PLATFORM STREQUAL "windows-mingw")
      add_custom_command(TARGET UpscaleDMDPlugin POST_BUILD
         COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/upscaledmd/libgcc_s_seh-1.dll"
         COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libstdc++-6.dll" "${PLUGINS_DIR}/upscaledmd/libstdc++-6.dll"
      )
   endif()
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(UpscaleDMDPlugin STATIC ${UPSCALEDMD_PLUGIN_SOURCES})

   target_include_directories(UpscaleDMDPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/upscaledmd
   )

   set_target_properties(UpscaleDMDPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-upscaledmd"
   )
endif()
