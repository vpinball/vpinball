# plugin CMake file to be included in main project CMakelists

# --- Inspector plugin ---

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 99)

set(RUNTIME_LIBS_DIR "${CMAKE_SOURCE_DIR}/third-party/runtime-libs/${PLATFORM}-${ARCH}")

set(INSPECTOR_PLUGIN_SOURCES
   plugins/inspector/inspector.cpp
   plugins/inspector/WebServer.h
   plugins/inspector/WebServer.cpp
   third-party/include/mongoose/mongoose.h
   third-party/include/mongoose/mongoose.c
)

if(BUILD_SHARED)
   add_library(InspectorPlugin MODULE ${INSPECTOR_PLUGIN_SOURCES})

   target_include_directories(InspectorPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/inspector
   )

   if(PLATFORM STREQUAL "windows" OR PLATFORM STREQUAL "windows-mingw")
      if(MSVC)
         set_target_properties(InspectorPlugin PROPERTIES
            MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>"
         )
         set(OPT_COMMON
            /Ob2
            /O2
            /Oi
            /fp:fast
            /fp:except-
            /Ot
            /GF
            /GS-
            /Gy
            /GR
            /Oy
            /GT
            /GL
            /w44005
            /Zc:__cplusplus
            /std:c++20
            $<$<NOT:$<STREQUAL:${ARCH},x64>>:/arch:SSE2>
         )
         target_compile_options(InspectorPlugin PUBLIC
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:CXX>:${OPT_COMMON}>>
            $<$<CONFIG:Release>:$<$<COMPILE_LANGUAGE:C>:${OPT_COMMON}>>
         )
         target_link_options(InspectorPlugin PUBLIC
            $<$<CONFIG:Release>:/INCREMENTAL:NO>
            $<$<CONFIG:Release>:/OPT:REF>
            $<$<CONFIG:Release>:/OPT:ICF>
            $<$<CONFIG:Release>:/LTCG>
         )
      endif()
      if(PLATFORM STREQUAL "windows-mingw")
         target_link_libraries(InspectorPlugin ws2_32)
      endif()
      set_target_properties(InspectorPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "$<IF:$<STREQUAL:${ARCH},x64>,plugin-inspector64,plugin-inspector>"
      )
   else()
      set_target_properties(InspectorPlugin PROPERTIES
         PREFIX ""
         OUTPUT_NAME "plugin-inspector"
      )
      if(PLATFORM STREQUAL "macos")
         set_target_properties(InspectorPlugin PROPERTIES
            SUFFIX ".dylib"
         )
      endif()
   endif()

   add_custom_command(TARGET InspectorPlugin POST_BUILD
      COMMAND "${CMAKE_COMMAND}" -E copy "${CMAKE_SOURCE_DIR}/plugins/inspector/plugin.cfg" "${PLUGINS_DIR}/inspector/plugin.cfg"
      COMMAND "${CMAKE_COMMAND}" -E copy "$<TARGET_FILE:InspectorPlugin>" "${PLUGINS_DIR}/inspector"
      COMMAND "${CMAKE_COMMAND}" -E rm "$<TARGET_FILE:InspectorPlugin>"
   )

   if(POST_BUILD_COPY_EXT_LIBS AND PLATFORM STREQUAL "windows-mingw")
      add_custom_command(TARGET InspectorPlugin POST_BUILD
         COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libgcc_s_seh-1.dll" "${PLUGINS_DIR}/inspector/libgcc_s_seh-1.dll"
         COMMAND "${CMAKE_COMMAND}" -E copy "${RUNTIME_LIBS_DIR}/libstdc++-6.dll" "${PLUGINS_DIR}/inspector/libstdc++-6.dll"
      )
   endif()
endif()

if(BUILD_STATIC)
   add_compile_options(-fPIC)

   add_library(InspectorPlugin STATIC ${INSPECTOR_PLUGIN_SOURCES})

   target_include_directories(InspectorPlugin PUBLIC
      ${CMAKE_SOURCE_DIR}/third-party/include
      plugins
      plugins/inspector
   )

   set_target_properties(InspectorPlugin PROPERTIES
      PREFIX ""
      OUTPUT_NAME "plugin-inspector"
   )
endif()
